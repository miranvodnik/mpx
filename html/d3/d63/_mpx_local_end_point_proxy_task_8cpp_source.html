42 void MpxLocalEndPointProxyTask::HandleLocalEndPointEvent (MpxEventBase *event)
43 {
44  MpxLocalEndPointEvent* localEndPointEvent = dynamic_cast <MpxLocalEndPointEvent*> (event);
45  if (localEndPointEvent == 0)
46  return;
47 
48  MpxLocalEndPoint* localEndPoint =
49  dynamic_cast <MpxLocalEndPoint*> ((MpxLocalEndPoint*) localEndPointEvent->endPoint ());
50  if (localEndPoint == 0)
51  return;
52 
53  switch (localEndPointEvent->flags ())
54  {
55  case EPOLLIN:
56  if ((localEndPointEvent->error () == 0) && (localEndPointEvent->size () != 0))
57  {
58  MpxEventStruct* req = new MpxEventStruct;
59  memset (req, 0, sizeof(MpxEventStruct));
60  if (localEndPoint->ReadXdrRequest ((xdrproc_t) xdr_MpxEventStruct, (void*) req) < 0)
61  {
62  delete req;
63  if (false)
64  cout << "proxy " << this << " event: local end point input error" << endl;
65  }
66  else
67  {
68  if (false)
69  cout << "proxy " << this << " event: local end point input ok" << endl;
70  xdrpair_t xdrPair = MpxXDRProcRegistry::Retrieve (req->m_code);
71  evnalloc_t evnAlloc = xdrPair.second.second;
72  if (evnAlloc == 0)
73  {
74  if (false)
75  cout << "proxy " << this << " missing event allocator" << endl;
76  }
77  else
78  {
79  MpxEventBase* event = evnAlloc ();
80  event->Decode (req);
81  Send (m_task, event, false);
82  }
83  xdr_free ((xdrproc_t) xdr_int, (char*) req);
84  }
85  }
86  break;
87  case EPOLLOUT:
88  if (false)
89  cout << "proxy " << this << " event: local end point output" << endl;
90  break;
91  default:
92  if (false)
93  cout << "proxy " << this << " event: local end point default" << endl;
94  break;
95  }
96 }
