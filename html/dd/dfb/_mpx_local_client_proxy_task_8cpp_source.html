41 void MpxLocalClientProxyTask::HandleLocalClientEvent (MpxEventBase *event)
42 {
43  MpxLocalClientEvent* localClientEvent = dynamic_cast <MpxLocalClientEvent*> (event);
44  if (localClientEvent == 0)
45  return;
46 
47  MpxLocalClient* localClient = dynamic_cast <MpxLocalClient*> ((MpxLocalClient*) localClientEvent->endPoint ());
48  if (localClient == 0)
49  return;
50 
51  switch (localClientEvent->flags ())
52  {
53  case EPOLLIN:
54  if ((localClientEvent->error () == 0) && (localClientEvent->size () != 0))
55  {
56  MpxEventStruct* req = new MpxEventStruct;
57  memset (req, 0, sizeof(MpxEventStruct));
58  if (localClient->ReadXdrRequest ((xdrproc_t) xdr_MpxEventStruct, (void*) req) < 0)
59  {
60  delete req;
61  if (false)
62  cout << "proxy " << this << " event: local client input error" << endl;
63  }
64  else
65  {
66  if (false)
67  cout << "proxy " << this << " event: local client input ok" << endl;
68  xdrpair_t xdrPair = MpxXDRProcRegistry::Retrieve (req->m_code);
69  evnalloc_t evnAlloc = xdrPair.second.second;
70  if (evnAlloc == 0)
71  {
72  if (false)
73  cout << "proxy " << this << " missing event allocator" << endl;
74  }
75  else
76  {
77  MpxEventBase* event = evnAlloc ();
78  event->Decode (req);
79  Send (m_task, event, false);
80  }
81  xdr_free ((xdrproc_t) xdr_int, (char*) req);
82  }
83  }
84  break;
85  case EPOLLOUT:
86  if (false)
87  cout << "proxy " << this << " event: local client output" << endl;
88  break;
89  default:
90  if (false)
91  cout << "proxy " << this << " event: local client default" << endl;
92  break;
93  }
94 }
