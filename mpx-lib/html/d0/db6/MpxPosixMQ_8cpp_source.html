152 int MpxPosixMQ::Start ()
153 {
154  m_mpx = (MpxTaskMultiplexer*) m_task->mpx ();
155 
156  if ((m_mpx == 0) || (m_mpx->getTid () != syscall (SYS_gettid)))
157  return -1;
158 
159  if ((m_ctx = m_mpx->ctx ()) == 0)
160  return -1;
161 
162  switch (m_type)
163  {
164  case PosixMQConsumerType:
165  m_IOHandle = m_ctx->RegisterDescriptor (EPOLLIN, m_fd, HandleMqConsumerIO, this, ctx_info);
166  break;
167  case PosixMQProviderType:
168  m_IOHandle = m_ctx->RegisterDescriptor (EPOLLOUT, m_fd, HandleMqProviderIO, this, ctx_info);
169  m_ctx->DisableDescriptor (m_IOHandle, EPOLLOUT);
170  break;
171  default:
172  return -1;
173  }
174  StartTimer (m_ctx->realTime ());
175  return 0;
176 }

267 int MpxPosixMQ::PostXdrRequest (xdrproc_t proc, void* data)
268 {
269  u_int dummy = 0;
270  u_int intSize = xdr_sizeof ((xdrproc_t) xdr_int, &dummy);
271  u_int msgSize = xdr_sizeof (proc, data);
272  u_int needSpace = intSize + msgSize;
273 
274  if (needSpace > m_msgSize)
275  return -1;
276 
277  char* chunk = new char [needSpace];
278  if (chunk == 0)
279  return -1;
280 
281  XDR xdr;
282 
283  xdrmem_create (&xdr, chunk, needSpace, XDR_ENCODE);
284  if (xdr_int (&xdr, (int*) &msgSize) != TRUE)
285  return -1;
286 
287  xdrmem_create (&xdr, chunk + intSize, needSpace - intSize, XDR_ENCODE);
288  if (proc (&xdr, data) != TRUE)
289  return -1;
290 
291  if (PutChunk ((u_char*) chunk) < 0)
292  {
293  delete [] chunk;
294  return -1;
295  }
296 
297  m_ctx->EnableDescriptor (m_IOHandle, EPOLLOUT);
298  return needSpace;

314 int MpxPosixMQ::PutChunk (u_char* chunk)
315 {
316  ssize_t usedSpace = m_chunkWriteIndex - m_chunkReadIndex;
317  if (usedSpace < 0)
318  usedSpace += m_chunkTableSize;
319 
320  if (usedSpace + 1 >= m_chunkTableSize)
321  {
322  ssize_t chunkTableSize = m_chunkTableSize;
323  chunkTableSize >>= 5;
324  ++chunkTableSize;
325  chunkTableSize <<= 5;
326 
327  u_char** chunkTable = new u_char* [chunkTableSize];
328  if (chunkTable == 0)
329  return -1;
330 
331  for (ssize_t index = 0; index < usedSpace; ++index)
332  chunkTable [index] = m_chunkTable [(m_chunkReadIndex + index) % m_chunkTableSize];
333 
334  delete [] m_chunkTable;
335  m_chunkTable = chunkTable;
336  m_chunkTableSize = chunkTableSize;
337  m_chunkReadIndex = 0;
338  m_chunkWriteIndex = usedSpace;
339  }
340 
341  ssize_t chunkIndex = m_chunkWriteIndex;
342  m_chunkTable [m_chunkWriteIndex++] = chunk;
343  m_chunkWriteIndex %= m_chunkTableSize;
344  return chunkIndex;
345 }
