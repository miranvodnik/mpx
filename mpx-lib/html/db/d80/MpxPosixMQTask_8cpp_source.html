162 void MpxPosixMQTask::PosixMQEventHandler (MpxEventBase* event)
163 {

165 
166  switch (posixMQEvent->flags ())

168  case 0:
169  break;

172  {
173  struct msg
174  {
175  MpxEventBase* event;
176  } msg = *((struct msg*) posixMQEvent->buffer ());
177 
178  msg.event->Invoke ();
179  delete msg.event;
180  return;
181  }
182  break;
183  case EPOLLOUT:
184  if ((posixMQEvent->error () == 0) && (posixMQEvent->size () != 0))
185  {
186  if (false)
187  cout << "POSIX MQ OUT" << endl;
188  return;
189  }
190  break;
