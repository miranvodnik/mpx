183 void MpxLocalMQTask::LocalEndPointEventHandler (MpxEventBase *event)
184 {
185  MpxLocalEndPointEvent* localEndPointEvent = dynamic_cast <MpxLocalEndPointEvent*> (event);
186  MpxLocalEndPoint* localEndPoint =
187  dynamic_cast <MpxLocalEndPoint*> ((MpxLocalEndPoint*) localEndPointEvent->endPoint ());
188 
189  if (false)
190  cout << "LOCAL END POINT EVENT: FLAGS = " << localEndPointEvent->flags () << ", ERROR = "
191  << localEndPointEvent->error () << ", SIZE = " << localEndPointEvent->size () << ", BUFFER = "
192  << ((void*) localEndPointEvent->buffer ()) << endl;
193  switch (localEndPointEvent->flags ())
194  {
195  case 0:
196  if (false)
197  cout << "LOCAL END POINT EVENT: TIMEOUT" << endl;
198  break;
199  case EPOLLIN:
200  if (false)
201  cout << "LOCAL END POINT EVENT: INPUT" << endl;
202  if ((localEndPointEvent->error () == 0) && (localEndPointEvent->size () != 0))
203  {
204  size_t size = localEndPointEvent->size ();
205  u_char* data = localEndPoint->fast () ? localEndPoint->FastRead (size) : localEndPointEvent->buffer ();
206  struct msg
207  {
208  MpxEventBase* event;
209  } msg;
210  for (; size > 0; size -= sizeof msg, data += sizeof msg)
211  {
212  msg = *((struct msg*) data);
213  if (msg.event->Invoke () != 0)
214  cout << "INVOKE FAILED" << endl;
215  delete msg.event;
216  }
217  return;
218  }
219  break;
220  case EPOLLOUT:
221  if (false)
222  cout << "LOCAL END POINT EVENT: OUTPUT" << endl;
223  if ((localEndPointEvent->error () == 0) && (localEndPointEvent->size () != 0))
224  return;
225  break;
226  default:
227  if (false)
228  cout << "LOCAL END POINT EVENT: UNKNOWN" << endl;
229  break;
230  }
231  if (false)
232  cout << "LOCAL END POINT ERASED" << endl;
233  epset::iterator it = m_epset.find (localEndPoint);
234  if (it != m_epset.end ())
235  m_epset.erase (it);
236  delete localEndPoint;
237 }
