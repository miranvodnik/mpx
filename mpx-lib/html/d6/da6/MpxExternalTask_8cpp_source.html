360 void MpxExternalTask::HandleLocalTaskQueryEvent (MpxEventBase* event)
361 {

363  if (localTaskQueryEvent == 0)
364  return;

367  if (srcTask == 0)

371  if (localClient->Connect (localTaskQueryEvent->path ()) != 0)
372  {

374  delete localClient;

377

382  localClient->PostXdrRequest ((xdrproc_t) xdr_MpxMessage, (void*) &msg);
383  m_lctmap [localClient] = srcTask;

385  if (true)

455 void MpxExternalTask::HandleLocalClientEvent (MpxEventBase* event)
456 {
457  MpxLocalClientEvent* localClientEvent = dynamic_cast <MpxLocalClientEvent*> (event);
458  if (localClientEvent == 0)
459  return;
460  MpxLocalClient* localClient = dynamic_cast <MpxLocalClient*> ((MpxLocalClient*) localClientEvent->endPoint ());
461  if (localClient == 0)
462  return;
463 
464  lctmap::iterator it = m_lctmap.find (localClient);
465  if (it == m_lctmap.end ())
466  return;
467 
468  MpxTaskBase* task = it->second;
469 
470  switch (localClientEvent->flags ())
471  {
472  case 0:
473  break;
474  case EPOLLIN:
475  if (false)
476  cout << "EXTERNAL TASK: LOCAL CLIENT EVENT IN" << endl;
477 
478  if ((localClientEvent->error () == 0) && (localClientEvent->size () != 0))
479  {
480  while (true)
481  {
482  MpxMessage* rpl = (MpxMessage*) malloc (sizeof(MpxMessage));
483  memset (rpl, 0, sizeof(MpxMessage));
484  if (localClient->ReadXdrRequest ((xdrproc_t) xdr_MpxMessage, rpl) < 0)
485  {
486  free (rpl);
487  break;
488  }
489  switch (rpl->m_Code)
490  {
491  case ExternalTaskReplyCode:
492  {
493  MpxExternalTaskReply* externalTaskReply = &(rpl->MpxMessage_u.m_externalTaskReply);
494  if ((MpxTaskBase*) externalTaskReply->task == 0)
495  break;
496 
497  MpxTaskMultiplexer* mpx = (MpxTaskMultiplexer*) this->mpx ();
498  m_lctmap.erase (it);
499  MpxLocalClientProxyTask* proxyTask = new MpxLocalClientProxyTask (
500  (MpxTaskBase*) externalTaskReply->task, localClient);
501  mpx->RegisterTask (proxyTask);
502  proxyTask->Send (proxyTask, new MpxStartEvent (), true);
503  proxyTask->Send (task, new MpxExternalTaskEvent (EPOLLIN, 0, 0, 0), false);
504  }
505  break;
506  default:
507  break;
508  }
509  xdr_free ((xdrproc_t) xdr_MpxMessage, (char*) rpl);
510  }
511  return;
512  }
513  break;
514  case EPOLLOUT:
515  if (false)
516  cout << "EXTERNAL TASK: LOCAL CLIENT EVENT OUT" << endl;
517 
518  if ((localClientEvent->error () == 0) && (localClientEvent->size () != 0))
519  {
520  Send (task, new MpxExternalTaskEvent (EPOLLOUT, 0, 0, 0), true);
521  return;
522  }
523  break;
524  default:
525  if (false)
526  cout << "EXTERNAL TASK: LOCAL CLIENT EVENT default" << endl;
527 
528  break;
529  }
530 
531  Send (task, new MpxExternalTaskEvent (localClientEvent->flags (), localClientEvent->error (), 0, 0), true);
532  m_lctmap.erase (it);
533  delete localClient;
534 }

536 void MpxExternalTask::HandleLocalEndPointEvent (MpxEventBase* event)
537 {

539  if (localEndPointEvent == 0)
540  return;
541 
542  MpxLocalEndPoint* localEndPoint =

545  return;

547  leset::iterator it = m_leset.find (localEndPoint);

553  case 0:

556 
557  break;
558  case EPOLLIN:
559  if (false)
560  cout << "EXTERNAL TASK: LOCAL END POINT EVENT IN" << endl;
561

569  {
570  free (req);
571  break;
572  }
573  switch (req->m_Code)
574  {
575  case ExternalTaskRequestCode:
576  {
577  MpxExternalTaskRequest* externalTaskRequest = &(req->MpxMessage_u.m_externalTaskRequest);
578  MpxTaskMultiplexer* mpx = (MpxTaskMultiplexer*) this->mpx ();
579  MpxEnvironment::BroadcastExternalEvent (mpx->externalTask (),
580  new MpxTaskQueryEvent (externalTaskRequest->taskName, TaskQueryEventLocal,
581  (void*) localEndPoint));
582  }
583  break;
584  default:
585  break;
586  }
587  xdr_free ((xdrproc_t) xdr_MpxMessage, (char*) req);
588  }
589  return;
590  }
591  break;
592  case EPOLLOUT:
593  if (false)
594  cout << "EXTERNAL TASK: LOCAL END POINT EVENT OUT" << endl;
595 
596  if ((localEndPointEvent->error () == 0) && (localEndPointEvent->size () != 0))
597  {
598  return;
599  }
600  break;
601  default:
602  if (false)
603  cout << "EXTERNAL TASK: LOCAL END POINT EVENT default" << endl;
604 
605  break;
606  }
607

609  delete localEndPoint;
610 }
